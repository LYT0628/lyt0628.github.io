4. Json转对象

public class Main{

  public static void main(String[] args) { 
    String json="{\"id\":666,\"firstName\":\"Li\",\"lastName\":\"Si\"}";
    var gson = new Gson();
    System.out.println(gson.fromJson(json, Employee.class));
  }
}
